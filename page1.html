#include<bits/stdc++.h>
using namespace std;
int main()
{
  int T;
  cin>>T;
  for(int t=1;t<=T;t++)
  {
    string a,b;
    cin>>a>>b;
    int ans=10000000;
    int n=a.size(),m=b.size();
    for(int i=0;i<b.size();i++)//x-1的部分 
    {
      int j=i;
      for(char c:a)
      if(j<m&&c==b[j])
      j++;
      
      ans=min(ans,m-(j-i)+n);
    }
    cout<<ans<<endl;
  }
}
