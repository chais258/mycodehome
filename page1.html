a,b,c分别表示第一个区间 ，第二个区间，第三个区间

例如ACB这种情况中

a表示A b表示C c表示B

a的左边界固定为1，枚举a的右边界，使得a的和>=1/3tot

接着以a的右边界作为b的左边界，向右枚举b的右边界，使得b的和>=1/3tot

剩下是c的区间

基本思想是尽量压缩左边的区间，使左边区间满足1/3要求的情况下尽量小，从而为右边部分留出空间

              
#include<bits/stdc++.h>

using namespace std;
const int N=2e5+10;
typedef long long ll; 
ll arr[4][N];
ll n;
bool isok=false;
struct Ans{
  ll id;
  ll l,r;
}ans[3];
bool cmp(Ans a,Ans b)
{
  return a.id<b.id;
}
bool check(ll a,ll b,ll c,ll sum)
{
  ll suma=0,sumb=0,sumc=0,lin;
  if(sum%3!=0)
  lin=sum/3+1;
  else
  lin=sum/3;
  for(ll i=1;i<=n;i++)//枚举a的右边界 
  {
    suma+=arr[a][i];
    if(suma>=lin)
    for(ll j=i+1;j<=n;j++)
    {
      sumb+=arr[b][j];

      if(sumb>=lin)
      {
        for(ll k=j+1;k<=n;k++)
        sumc+=arr[c][k];
        if(sumc>=lin)
        {
          ans[0]={a,1,i},ans[1]={b,i+1,j},ans[2]={c,j+1,n};
          sort(ans,ans+3,cmp);
          for(ll m=0;m<3;m++)
          cout<<ans[m].l<<" "<<ans[m].r<<" ";
          cout<<endl;
          return true;
        }
        else
        return false;
      }
      if(j==n)
      return false;
    }
  }
  return false;
}
void solved()
{
  
  cin>>n;
  ll sum=0;
  for(ll i=1;i<=3;i++)
  for(ll j=1;j<=n;j++)
  {
    cin>>arr[i][j];
    sum+=arr[i][j];
  }
  sum/=3;
  for(ll i=1;i<=3;i++)
  for(ll j=1;j<=3;j++)
  for(ll k=1;k<=3;k++)
  {
    if(i==j||j==k||i==k)
    continue;
    if(check(i,j,k,sum))
    return;
  }
  cout<<-1<<endl;
  
}
int main()
{
  ll t;
  cin>>t;
  while(t--)
  {
    solved();
  }
}
